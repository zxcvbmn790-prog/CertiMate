# 1
-- 세팅 과정 --
사용하고자 하는 파일에 "우클릭 + Shift"
Powershell로 열기
git config --global user.email "내 이메일"
git config --global user.name "내 이름"

-- Git 기본 세팅 --
작업폴더에서 git 쓰고 싶으면 "git init"부터 입력 -> Git이라는 소프트웨어가 작업을 감시하기 시작함

-- Git 기록 방법 --
git add 파일명 => 파일 현재 상태를 기록하는 방법, 기록할 파일 고르기
git commit -m '메모' => 고른 파일 기록 명령
# 버전 생성이라고도 한다

-- Sturucture --
[작업폴더] -git add(스테이징한다)-> [staging area] -git commit-> [repository]

-- Staging Bonus --
git add 파일1 파일2 => (공백으로 구분)
git add . =>  현재 작업 폴더의 모든 파일 스테이징

-- 추가 명령어 --
git status => 상태창 열기 : 어떤 파일들이 스테이징 되어있는지 수정되었는지를 알려준다
git log --all --oneline : 커밋한 기록을 보기 + 커밋아이디도 볼 수 있다

#2
-- VS Code 전용 git 방법 --
왼쪽에 트리모양 아이콘
변경시 Changes가 나온다
포커스 대고 + : add
- : add 취소
위에 체크 표시 : commit

-- 차이 비교 --
git diff => 최근 commit vs 현재 파일 차이점을 보여준다 (j, k로 스크롤. q로 나가기)
    단점 => 터미널의 한계로 보기 복잡, code가 복잡해지면 보기가 힘들다
git difftool => vim에디터로 더욱 시각적으로 좋게 차이점을 비교해준다 (hjkl이 방향키. :q혹은 :qa로 종료)
git difftool 커밋아이디 => 현재파일과 특정 커밋과의 차이점 비교 가능(두개까지 공백구분으로 비교 가능)
