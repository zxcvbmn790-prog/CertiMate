-- 세팅 과정 --
사용하고자 하는 파일에 "우클릭 + Shift"
Powershell로 열기
git config --global user.email "내 이메일"
git config --global user.name "내 이름"

-- Git 기본 세팅 --
작업폴더에서 git 쓰고 싶으면 "git init"부터 입력 -> Git이라는 소프트웨어가 작업을 감시하기 시작함

-- Git 기록 방법 --
git add 파일명 => 파일 현재 상태를 기록하는 방법, 기록할 파일 고르기
git commit -m '메모' => 고른 파일 기록 명령
# 버전 생성이라고도 한다

-- Sturucture --
[작업폴더] -git add(스테이징한다)-> [staging area] -git commit-> [repository]

-- Staging Bonus --
git add 파일1 파일2 => (공백으로 구분)
git add . =>  현재 작업 폴더의 모든 파일 스테이징

-- 추가 명령어 --
git status => 상태창 열기 : 어떤 파일들이 스테이징 되어있는지 수정되었는지를 알려준다
git log --all --oneline : 커밋한 기록을 보기

-- VS Code 전용 git 방법 --
왼쪽에 트리모양 아이콘
변경시 Changes가 나온다
포커스 대고 + : add
- : add 취소
위에 체크 표시 : commit
